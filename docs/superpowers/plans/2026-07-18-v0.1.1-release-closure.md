# GoAgents Host Lifecycle v0.1.1 Release Closure Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将已进入 `main` 的 Host 生命周期硬化收敛为一个经过完整本机、真实 Provider、远端 CI 和外部消费者验证的不可变多模块发布。

**Architecture:** release candidate 修改逐模块版本门禁、可复用的仓库外 consumer 验证助手和当前状态文档；2026-07-19 批准的审查修正另允许最小修改直接验收 test/harness，以强化 exact SHA 和随机 sentinel 门禁，但不修改运行时代码。候选通过独立复审后推送 `main`，三个 annotated tags 以一次 atomic push 固定到同一 commit；正式 tag consumer 通过后创建单一 GitHub Release，最后用新的未打 tag 提交回写发布证据。

**Tech Stack:** Go 1.26.1、Bash、Git、GitHub CLI、GitHub Actions、macOS Keychain、SQLite、OpenAI-compatible Provider。

所有 clean-worktree 门禁统一使用以下 helper。可选参数是目标仓库路径，省略时使用当前目录；
每个调用它的独立 shell Step 必须在同一 shell 内先逐字定义该 helper，不能依赖其他 Step
遗留的函数状态。底层 `git status` 失败和非空工作区分别只输出固定标签，不输出目标路径或
Git 原始 stderr：

```bash
require_clean_worktree() {
  local repo_path="${1:-.}"
  local worktree_status
  local git_status

  if worktree_status="$(
    git -C "$repo_path" status --short --untracked-files=all 2>/dev/null
  )"; then
    git_status=0
  else
    git_status=$?
  fi
  if (( git_status != 0 )); then
    printf 'release gate blocked: worktree status unavailable\n' >&2
    return 1
  fi
  if [[ -n "$worktree_status" ]]; then
    printf 'release gate blocked: worktree is not clean\n' >&2
    return 1
  fi
}
```

允许重试的只读 GitHub/Git 查询统一在对应 Step 的同一个 shell 中使用以下助手；调用方只有在
“成功但新资源列表暂为空”时才可显式调用 `retry_pause`。助手不把 404、401/403 或内容不一致
归为瞬时错误，也不会隐藏最终失败：

```bash
set -euo pipefail
retry_deadline=$((SECONDS + 300))

retry_pause() {
  local remaining=$((retry_deadline - SECONDS))
  (( remaining > 0 )) || return 1
  (( remaining > 30 )) && remaining=30
  sleep "$remaining"
}

retry_external_read() {
  local error_file
  error_file="$(mktemp "${TMPDIR:-/tmp}/goagents-read.XXXXXX")"

  while :; do
    if "$@" 2>"$error_file"; then
      rm -f "$error_file"
      return 0
    fi

    if ! grep -Eiq \
      'connection reset|timed out|timeout|unexpected EOF|HTTP (502|503|504)|status code (502|503|504)|could not resolve host|failed to connect|remote end hung up unexpectedly|TLS handshake timeout' \
      "$error_file"; then
      printf 'non-retryable external read failure\n' >&2
      sed -n '1,10p' "$error_file" >&2
      rm -f "$error_file"
      return 1
    fi

    if ! retry_pause; then
      printf 'external read retry budget exhausted\n' >&2
      sed -n '1,10p' "$error_file" >&2
      rm -f "$error_file"
      return 1
    fi
    printf 'retrying transient external read after at most 30 seconds\n' >&2
  done
}
```

```bash
expected_internal_requirements() {
  local owner="$1"
  local spec
  local dir
  local module
  local version

  for spec in "${internal_requirements[@]}"; do
    IFS='|' read -r dir module version <<<"$spec"
    if [[ "$dir" == "$owner" ]]; then
      printf '%s %s\n' "$module" "$version"
    fi
  done
}

actual_internal_requirements() {
  local go_mod="$1"

  awk -v prefix="$MODULE_PREFIX" '
    BEGIN { mode = "" }
    $1 == "require" && $2 == "(" { mode = "require"; next }
    $1 == "replace" && $2 == "(" { mode = "replace"; next }
    $1 == ")" { mode = ""; next }
    $1 == "require" && index($2, prefix) == 1 {
      print $2, $3
      next
    }
    mode == "require" && index($1, prefix) == 1 {
      print $1, $2
    }
  ' "$go_mod"
}

check_internal_require_versions() {
  local dir="$1"
  local difference

  if ! difference="$(diff -u \
    <(expected_internal_requirements "$dir" | LC_ALL=C sort) \
    <(actual_internal_requirements "$ROOT/$dir/go.mod" | LC_ALL=C sort))"; then
    report_set_difference "$dir internal requirements" "$difference"
  fi
}
```

```bash
check_release_delta_tags() {
  local spec
  local dir
  local module
  local version
  local tag
  local release_status
  local difference
  local -a manifest_delta_tags=()

  for spec in "${published_modules[@]}"; do
    IFS='|' read -r dir module version tag release_status <<<"$spec"
    case "$release_status" in
      existing)
        ;;
      release-delta)
        manifest_delta_tags+=("$tag")
        ;;
      *)
        report_error "$dir: unknown release status: $release_status"
        ;;
    esac
  done

  if ! difference="$(diff -u \
    <(printf '%s\n' "${release_delta_tags[@]}" | LC_ALL=C sort -u) \
    <(printf '%s\n' "${manifest_delta_tags[@]}" | LC_ALL=C sort -u))"; then
    report_error "release delta set mismatch"
    printf '%s\n' "$difference" >&2
  fi
}
```
